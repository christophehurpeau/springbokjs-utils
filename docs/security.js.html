var crypto = require('crypto');

/**
 * Security
 * Only for NodeJS
 *
 * @module security
 */

/**
 * Create an md5 hash
 *
 * @param {Buffer|String} data
 * @return {String} base64 string
 */
exports.md5 = function(data) {
    var shasum = crypto.createHash('md5');
    shasum.update(data);
    return shasum.digest('base64');
};
